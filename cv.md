function dirReduc(arr){
  const directionPairs = {
    "NORTH": "SOUTH",
    "SOUTH": "NORTH", 
    "EAST": "WEST",
    "WEST": "EAST",
  }
  const result = [];

  for (const direction of arr) {
    if (result[result.length - 1] === directionPairs[direction]) {
      result.pop(); 
    } else {
      result.push(direction);
    }
  }

  return result;
}
