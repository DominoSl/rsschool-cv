Directions Reduction task from codewars 

              
function dirReduc(arr){
  const directions = {
    "NORTH": "SOUTH",
    "SOUTH": "NORTH", 
    "EAST": "WEST",
    "WEST": "EAST",
  }
    
  return arr.reduce((result, direction) => {
    if (result.length === 0) {
      return [direction];
    }    

    const lastDirection = result[result.length - 1];
    
    if (lastDirection === directions[direction]) {
      result.length--;
      
      return result;
    } else {
      return [...result, direction];
    }
  }, []);
}
